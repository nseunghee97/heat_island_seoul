extract_tag <- function(category_num){
  df <- subset.data.frame(KRvideos, (category_id==category_num)&(tags!='[none]'))
  keywords_vector <- vector()
  for (tag in df$tags){
    keywords <- tag %>%
      tolower() %>%
      str_remove_all('"') %>%
      str_replace_all(" ", "\\|") %>%
      str_split("\\|") %>%
      unlist() %>%
      unique()
    keywords_vector <- append(keywords_vector, keywords)
  }
}
